// Reglas de Firestore para el Test DISC
// Copia estas reglas en la consola de Firebase > Firestore > Reglas

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permitir lectura y escritura para usuarios autenticados
    match /disc/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Regla más permisiva para desarrollo (solo para testing)
    // CUIDADO: No usar en producción
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

// 1. Ve a https://console.firebase.google.com/
// 2. Selecciona tu proyecto "disceneagram"
// 3. Ve a Firestore Database > Reglas

// 5. Haz clic en "Publicar"
